# 路径别名

  • replacement 后面得手动拼接上 “/”
// /@/views/ 默认转成 ./srcviews
export default {
  resolve: {
    alias: [
      {
        find: /^\/@\//,
        replacement: `${resolve(__dirname, "./src")}/`,
      },
    ],
  },
};

  "types": ["vite/client"]
