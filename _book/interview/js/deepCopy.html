function deepCopy(obj) { const newObj = Array.isArray(obj) ? [] : {} for (let key in obj) { const item = obj[key] if (typeof item !== 'object') { newObj[key] = item } else if (item instanceof Date) { newObj[key] = new Date(item.getTime()) } else if (item instanceof RegExp) { newObj[key] = new RegExp(item) } else if (item.nodeType === 1) { newObj[key] = item.cloneNode(true) } else { newObj[key] = deepCopy(item) } } return newObj } const obj = { str: 'string', date: new Date(), regexp: new RegExp(/a/), body: document.body, arr: [1,2,3], json: { a: 1, }, } const newObj = deepCopy(obj) console.log(obj, newObj) console.log(obj.str === newObj.str) console.log(obj.date === newObj.date) console.log(obj.regexp === newObj.regexp) console.log(obj.body === newObj.body) console.log(obj.arr === newObj.arr) console.log(obj.json === newObj.json)
